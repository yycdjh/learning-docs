# 双指针——leetcode 977

```javascript
function sortedSquares(nums: number[]): number[] {
  let result = [];
  let i = 0;
  let j = nums.length - 1;
  let k = nums.length - 1;

  while (i <= j) {
    if (nums[i] * nums[i] > nums[j] * nums[j]) {
      result[k] = nums[i] * nums[i];
      k--;
      i++;
    } else {
      result[k] = nums[j] * nums[j];
      k--;
      j--;
    }
  }

  return result;
}
```
