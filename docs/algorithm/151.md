# 反转字符串中的单词

```javascript
function reverseWords(s: string): string {
  let strArr: string[] = s.split("");
  removeSpaces(strArr);
  let length: number = strArr.length;
  // 翻转整个字符串
  reverseWords1(strArr, 0, length - 1);

  // 翻转单个单词
  let start: number = 0;
  let end: number = 0;
  while (start < length) {
    end = start;
    while (strArr[end] !== " " && end < length) {
      end++;
    }
    reverseWords1(strArr, start, end - 1);
    start = end + 1;
  }

  return strArr.join("");
}

function removeSpaces(s: string[]) {
  let slow = 0;
  for (let fast = 0; fast < s.length; fast++) {
    if (s[fast] !== " ") {
      if (slow !== 0) {
        s[slow] = " ";
        slow++;
      }
      while (fast < s.length && s[fast] !== " ") {
        s[slow] = s[fast];
        fast++;
        slow++;
      }
    }
  }
  s.length = s[slow - 1] === "" ? slow - 1 : slow;
}

function reverseWords1(s: string[], start: number, end: number) {
  let left = start;
  let right = end;
  while (left < right) {
    let temp = s[right];
    s[right] = s[left];
    s[left] = temp;
    left++;
    right--;
  }
}
```
