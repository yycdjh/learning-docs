# 用队列实现栈 ——leetcode 225

```javascript
class MyStack {
    private queue: number[]
    constructor() {
        this.queue = []
    }

    push(x: number): void {
        this.queue.push(x)
    }

    pop(): number {
       for(let i = 0; i < this.queue.length -1; i++){
            this.queue.push(this.queue.shift()!)
       }
       return this.queue.shift()
    }

    top(): number {
        let res: number = this.pop()
        this.push(res)
        return res
    }

    empty(): boolean {
        return this.queue.length === 0
    }
}

 * var param_3 = obj.top()
 * var param_4 = obj.empty()
```
