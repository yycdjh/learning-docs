# 删除字符串中的所有相邻重复项 —— leetcode 1047

```javascript
function removeDuplicates(s: string): string {
  let stack: string[] = [];

  for (let i = 0; i < s.length; i++) {
    if (s[i] === stack[stack.length - 1]) {
      stack.pop();
    } else {
      stack.push(s[i]);
    }
  }

  return stack.join("");
}
```
